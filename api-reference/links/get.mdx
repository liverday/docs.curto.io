---
title: "Listar todos os links"
api: "GET https://api.curto.io/v1/links"
description: "Esse endpoint lista todos os links curtos"
---

### Parâmetros de Query

<ParamField body="sortBy" type="enum">
  Ordena os links por data de criação ou pela quantidade de cliques: "date" ou "performance"
</ParamField>

<ParamField body="sortDirection" type="enum">
  A direção da ordenação. "asc" para crescente e "desc" para decrescente.
</ParamField>

<ParamField body="page" type="number">
  Número da página que você está buscando os links. Se não for enviado, o padrão será 1
</ParamField>

<ParamField body="pageSize" type="number">
  A quantidade de itens na página. Se não for enviado, o padrão será 12, e o valor máximo para esse parâmetro é 100.
</ParamField>

### Resposta

<ResponseField name="status" type="string">
  Indica se a requisição foi bem sucedida. OK para quando for bem sucedida, e ERROR para quando não for.
</ResponseField>

<ResponseField name="pagination" type="object">
  Objeto com os dados atuais de paginação.

  <Expandable title="Pagination Object">
    <ResponseField name="current" type="number">
      O número da página atual
    </ResponseField>

    <ResponseField name="previous" type="number">
      O número da página anterior
    </ResponseField>

    <ResponseField name="current" type="number">
      O número da próxima página
    </ResponseField>

    <ResponseField name="total_pages" type="number">
      O número total de páginas
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total_rows" type="number">
  Indica o número de linhas retornadas na página atual.
</ResponseField>

<ResponseField name="data" type="array">

  Uma lista de links.

  <Expandable title="Link Object">

    <ResponseField name="id" type="string">
      Esse é o ID interno do link criado. Você não precisa memorizar esse dado
    </ResponseField>

    <ResponseField name="destination" type="string">
      É o destino do seu link curto (o link longo).
    </ResponseField>

    <ResponseField name="token" type="string">
      O identificador do seu link curto.
    </ResponseField>

    <ResponseField name="short_link" type="string">
      É o endereço compartilhável do seu link curto. Você pode copiar esse valor e compartilhar com sua audiência.
    </ResponseField>

    <ResponseField name="domain" type="string">
      Esse é o domínio utilizado para criar seu link curto. 
    </ResponseField>

    <ResponseField name="domain_id" type="string">
      Esse é o ID interno do domínio utilizado para criar seu link curto. Você não precisa memorizar esse valor.
    </ResponseField>

    <ResponseField name="password" type="boolean">
      Indica se o link criado é protegido por senha ou não.
    </ResponseField>

    <ResponseField name="created_at" type="timestamp">
      Data de criação do link curto.
    </ResponseField>

    <ResponseField name="updated_at" type="timestamp">
      Data da última atualização de um link curto.
    </ResponseField>

    <ResponseField name="clicks" type="number">
      O número de cliques que esse link curto já teve.
    </ResponseField>

  </Expandable>

</ResponseField>

```bash
curl -H 'X-Curto-Api-Key: <your_key>' \
https://api.curto.io/v1/links
```
